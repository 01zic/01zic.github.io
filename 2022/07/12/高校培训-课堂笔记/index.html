                    url:url,
                    success:res=>{
                        $("#container").html("");
                        const songs = res.result.songs;
                        songs.forEach(item=>{
                            var template =`
                            <div>
                             <p>${item.name}</p>
                             <p>${item.id}</p>
                            <div>
                            `
                            $("#container").append(template)
                        })
                    }
                })
            }
        }
    </script>

<!DOCTYPE html>

<head>

            height: 50px;

        }

            height: 2000px;
        }
    </style>
</head>

<body>
    <div id="nav" style="opacity:0 ;"></div>
    <!-- BOM:操作浏览器对象模型的标注(browser object model) -->
    <!-- 
        a、滚动条距离顶部300px,让导航条显示
        b、小于300px,将其慢慢隐藏,0完全隐藏
     -->
    <script>
        /* 1、给窗口绑定滚动事件 */
        var nav = document.getElementById("nav");
        window.onscroll = function(){
            // 2、获取滚动条距离顶部的height
            var scrollTop = document.documentElement.scrollTop;
            /* 到达300px,让opacity==1 */
            var opacity = scrollTop/300;
            console.log(opacity)
            // 3、设置nav的透明度
            nav.style.opacity = opacity;

        }
    </script>

</body>

隐藏和显示的例子


<body>
    <div id="content" style="display: block;">显示/隐藏</div>
    <button id="btn">切换</button>
    <script>
        /* 先清楚业务,才有代码 */
        /* 需求:点击button,让div切换隐藏和显示的状态 */
        /* 实现:1、获取btn */
        const btn = document.getElementById("btn");
        var content = document.getElementById("content")
        /* 2、对其执行点击事件 */
        btn.onclick = function () {
            /* 3、if 如果div显示,就将其隐藏,如果隐藏就将其显示 */
            /* 3-1、需要获取div的状态 */
            let state = content.style.display;
            console.log(state)
            /* 3-2、if判断 */
            if(state=="block"){
                content.style.display="none"
            }else{
                content.style.display = "block"
            }
        }
    </script>
</body>
